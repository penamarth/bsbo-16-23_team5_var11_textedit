# Диаграмма прецедентов UML для текстового редактора

```mermaid
classDiagram
  class TextEditor {
    -__document: Document
    -__text: String
    +inputText(char : string)
    +insertText(character: Character)
    +deleteText(character: Character)
    +saveState(): Snapshot
    +restoreState(snapshot: Snapshot)
    +copyText(start: int, end: int)
    +pasteText(text: String, cursor: Cursor)
  }

  class Document {
    -__content: List~Character~
    +getContent(): List~Character~
    +setContent(content: List~Character~)
  }

  class Character {
    -__text: char
    -__position: int
    +getText(): char
    +getPosition(): int
    +setPosition(position: int)
  }

  class Cursor {
    -__position: int
    +getPosition(): int
    +setPosition(position: int)
    +move(offset: int)
  }

  class Snapshot {
    -content: List~Character~
    -cursorPosition: int
    +getContent(): List~Character~
    +getCursorPosition(): int
  }

  class Caretaker {
    -snapshots: List~Snapshot~
    +addSnapshot(snapshot: Snapshot)
    +getSnapshot(index: int): Snapshot
  }

  TextEditor "1" -- "1" Document : composition
  Document "1" -- "*" Character : composition
  TextEditor "1" --> "1" Cursor : dependency
  TextEditor "1" -- "1" Caretaker : composition
  Caretaker "1" -- "*" Snapshot : composition
```
