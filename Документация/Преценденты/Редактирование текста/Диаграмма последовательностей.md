# Диаграмма последовательностей для редактировования текста

```mermaid
sequenceDiagram
    participant User
    participant Editor
    participant Document
    participant Paragraph
    participant Sentence
    participant Word

    Note over User,Word: Вставка: "\nПривет, мой друг.\nпишу тебе письмо"
    User->>Editor: insertText("\nПривет, мой друг.\nпишу тебе письмо")
    Editor->>Document: insertText("\nПривет, мой друг.\nпишу тебе письмо")
    
    Document->>Paragraph: new ParagraphComponent()
    Document->>Sentence: new SentenceComponent()
    Document->>Word: new WordComponent("Привет,")
    Word->>Sentence: add(WordComponent)
    Document->>Word: new WordComponent("мой")
    Word->>Sentence: add(WordComponent)
    Document->>Word: new WordComponent("друг.")
    Word->>Sentence: add(WordComponent)
    Sentence->>Paragraph: add(SentenceComponent)
    Paragraph->>Document: add(ParagraphComponent)
    
    Document->>Paragraph: new ParagraphComponent()
    Document->>Sentence: new SentenceComponent()
    Document->>Word: new WordComponent("пишу")
    Word->>Sentence: add(WordComponent)
    Document->>Word: new WordComponent("тебе")
    Word->>Sentence: add(WordComponent)
    Document->>Word: new WordComponent("письмо")
    Word->>Sentence: add(WordComponent)
    Sentence->>Paragraph: add(SentenceComponent)
    Paragraph->>Document: add(ParagraphComponent)

```
